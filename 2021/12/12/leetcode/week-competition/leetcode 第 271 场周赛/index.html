# 环和杆

题目链接：环和杆

题目详情:

该题是明显的计数题，记录每一条杆上不同颜色的环的数量，当三种颜色数量都不为零时，总个数加一

# 解法一

利用数组

class Solution {
public:
    int countPoints(string rings) {
        vector<vector<int>>a(10,vector<int>(3));     // 总共最多有 10 个杆，a [i][0],a [i][1],a [i][2]
		                                             // 分别记录 R、G、B 的个数
        for(int i=0;i<rings.length();i+=2)
        {
            if(rings[i]=='R')
            {
                a[rings[i+1]-48][0]++;
            }
            else if(rings[i]=='G')
            {
                a[rings[i+1]-48][1]++;
            }
            else if(rings[i]=='B')
            {
                a[rings[i+1]-48][2]++;
            }
        }
        int ans=0;
        for(int i=0;i<=9;i++)
        {
            if(a[i][0]&&a[i][1]&&a[i][2])     // 当三种杆数量都不为 0 时，ans 加一
            ans++;
        }
        return ans;
    }
};

# 解法二

下面为周赛用时最少的 Aoxiang Cui 的代码

class Solution {
public:
    int countPoints(string s) {
        map<char, int> A;
        A['R'] = 0;
        A['G'] = 1;
        A['B'] = 2;
        vector<int> cnt(10);
        for (int i = 0; i < (int)s.size(); i += 2) {
            int x = A[s[i]];
            int y = s[i + 1] - '0';
            cnt[y] |= 1 << x;
        }
        int ret = 0;
        for (int k = 0; k < 10; ++k) {
            if (cnt[k] == 7) ++ret;
        }
        return ret;
        
    }
};

借鉴经验：

在该题中我的方法中的若找到某个字母，则在对应的数组位置加一，最后需判断三个代表 环的元素是否大于零，需要用 if 来判断，缺点是单判断的个数变多是，代码则会显得冗长 而在其所用的是二进制的计数法，

因为二进制的特点为 0，1 组合，那么这样可以把不同的位看作是一个一个的容器，当存储某个东西时，可将该容器值变为 1

该方法中的核心代码便是：

int x = A[s[i]];
int y = s[i + 1] - '0';
cnt[y] |= 1 << x;

这几行代码表示的意思分别为

第一行：对读到的环进行判断，依据之前的 map，找到对应容器

第二行：对读到的杆编号转化为 int 类型

第三行：将该环存入，1<<x 表示的为环所在位置

那么当所有环都存完后，那么如果三个环都存入，那么 cnt[i] 的值必然是 7 ，

因为在该方法中，由 map，R、G、B 对应的位置分别为20,21,222^0,2^1,2^220,21,22, 如果都存入，三个

位置显示都为111，那么cnt[i]=20+21+22=7cnt[i]=2^0+2^1+2^2=7cnt[i]=20+21+22=7

所以结尾处代码为

for (int k = 0; k < 10; ++k) 
{
     if (cnt[k] == 7) ++ret;
}

# 子数组范围和

题目链接：子数组范围和

题目详情:

方法：滑动窗口 (暴力)

i 遍历每一个数组开头，开始增加窗口长度之前，先更新 MAX 和 MIN 的值，之后 j 从 i+1 开始不断增加 增加的过程中不断更新 MAX ， MIN 的值，更新完后加上最大值减去最小值的值，如此，循环结束后 自然能够得到答案

class Solution {
public:
    long long subArrayRanges(vector<int>& nums) {
        long long MAX,MIN,ans=0;
        for(int i=0;i<nums.size();i++)
        {
            MAX=MIN=nums[i];
            for(int j=i+1;j<nums.size();j++)
            {
                MAX=fmax(MAX,nums[j]);
                MIN=fmin(MIN,nums[j]);
                ans+=MAX-MIN;      
            }
        }
        return ans;
    }
};

# 给植物浇水 II

题目链接：给植物浇水 II

题目详情:

# 方法一

我的思路：题目中要求求重新灌满水的次数，由题可知，两人浇水过程的位置是对称的， 所以说浇完后两人的次数要么相等要么相差 1 ，那么每一位置的浇水量必然不大于该人的水罐容量，也就不需要考虑某人浇不了水要等另一个人浇 (这样就明显变为另一个题目了)

(或许可以有这样的题目: "如果两人分别从两边开始浇水，若可以浇水，则浇，若水量不够则重新灌水，若还是不够，则选择原地等待，等另一个人浇。请返回以下值，若可以全浇完，返回重新灌水次数，若不行，返回 - 1")

接上面思路：

这样我们可以知道，每个人浇水次数至少是植物数整除 2 的结果，所以我们可以用循环循环这一次数，结束时，根据提示， plants[i]>0 ，所以可在浇完水后将浇完水的植物 plant[i] 赋值 0 ，最后只需判断最中间

的 plants[i] 是否为 0 ，若不为 0 ，则还需再浇一次，因此判断两人剩余量的最大值是否可行，不可行则灌水次数加一

据上面分析，我们所需要的变量为： curA 记录 Alice 剩余水量， curB 记录 Bob 剩余水量， ans 记录总灌水次数， time 记录最少浇水次数

以下为代码:

class Solution {
public:
    int minimumRefill(vector<int>& plants, int capacityA, int capacityB) {
        int times=plants.size()/2;
        int curA=capacityA,curB=capacityB;
        int ans=0;
        for(int i=0;i<times;i++)
        {
            if(curA>=plants[i])
            {
                curA-=plants[i];
                plants[i]=0;
            }
            else
            {
                ans++;
                curA=capacityA-plants[i];
                plants[i]=0;
            }
            if(curB>=plants[plants.size()-1-i])
            {
                curB-=plants[plants.size()-1-i];
                plants[plants.size()-1-i]=0;
            }
            else
            {
                ans++;
                curB=capacityB-plants[plants.size()-1-i];
                plants[plants.size()-1-i]=0;
            }
        }
        if(plants[times])
        {
            if(max(curA,curB)<plants[times])
            ans++;
        }
        return ans;
    }
};

class Solution {
public:
    int minimumRefill(vector<int>& plants, int capacityA, int capacityB) {
        int n = plants.size(), ans = 0, a = capacityA, b = capacityB;
        for (int i = 0, j = n - 1; i < j; ++i, --j) {
            if (a < plants[i]) a = capacityA, ++ans;
            a -= plants[i];
            if (b < plants[j]) b = capacityB, ++ans;
            b -= plants[j];
        }
        if (n & 1 && max(a, b) < plants[n / 2]) ++ans;
        return ans;
    }
};

# 摘水果

题目链接：摘水果

class Solution {
public:
    int maxTotalFruits(vector<vector<int>>& fruits, int startPos, int k) {
        int n = fruits.size();
        
        // 前缀和，用于求区间 [l, r] 共有多少水果
        vector<int> sum;
        sum.push_back(0);
        for (int i = 1; i <= n; ++ i)
            sum.push_back(fruits[i - 1][1] + sum[i - 1]);

        vector<int> pos;
        for (int i = 0; i < n; ++ i)
            pos.push_back(fruits[i][0]);

        // 枚举可能抵达的区间
        int ans = 0;
        for (int x = k; x >= 0; -- x)
        {
            int y = (k - x) / 2;
            int l, r;
            // x + 2y = k
            l = startPos - x, r = startPos + y;
            auto pl = lower_bound(pos.begin(), pos.end(), l) - pos.begin(), pr = upper_bound(pos.begin(), pos.end(), r) - pos.begin();
            ans = max(ans, sum[pr] - sum[pl]);
            // 2y + x = k
            l = startPos - y, r = startPos + x;
            pl = lower_bound(pos.begin(), pos.end(), l) - pos.begin(), pr = upper_bound(pos.begin(), pos.end(), r) - pos.begin();
            ans = max(ans, sum[pr] - sum[pl]);
        }
        
        return ans;
    }
};
