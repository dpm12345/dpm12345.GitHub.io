# pos 数组记录每个位置的剩余油量

  1. 使用另一个数组pospospos，其中pos[i]=gas[i]−cost[i]pos[i]=gas[i]-cost[i]pos[i]=gas[i]−cost[i], 从下标000 开始遍历，若pos[i]<0pos[i]<0pos[i]<0, 则必不以此为起点；若 pos[i]>=0pos[i]>=0pos[i]>=0, 则以其为起点，

    用resresres 存储当前剩余油量，之后以此为起点遍历之后位置nnn，若pos[n]+res<0pos[n]+res<0pos[n]+res<0, 此起点不满足题意，若 >=0>=0>=0, 则更新resresres，并使

    n=(n+1)n=(n+1)%pos.size()n=(n+1)，继续遍历若正常结束循环，则可以返回iii 的值

以下为代码

class Solution {
public:
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
        if(gas.size()==1)return (gas[0]<cost[0])?-1:0;
        vector<int>pos(gas.size());
        for(int i=0;i<pos.size();i++)
        {
            pos[i]=gas[i]-cost[i];
        }
        for(int i=0;i<pos.size();i++)
        {
            if(pos[i]>0)
            {
                int res=pos[i],n=(i+1)%pos.size();
                while(n!=i)
                {
                    if(pos[n]+res<0)
                    break;
                    res+=pos[n];
                    n=(n+1)%pos.size();
                }
                if(n==i)return i;
            }
            
        }
        return -1;
    }
};

  1. 优化：由上代码，当不满足pos[n]+res>=0pos[n]+res>=0pos[n]+res>=0 时，nnn 即为以iii 为起点不可能到达的位置，那么这两个位置之间的位置都不可能到达nnn 因为以iii 为起点，当它到两者中间位置时，初始油量大于等于零，而从iii 位置到达不了，更别说两者其中位置了由此，可增加一个变量记录每次前进的长度numnumnum，当以某一起点得不到答案时，使i=i+num+1i=i+num+1i=i+num+1 更新iii，提高效率

    class Solution {
public:
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
        int n=gas.size();
        int i=0;
        while(i<n)
        {
            int num=0,total_gas=0,total_cost=0;
            while(num<n)
            {
                int j=(i+num)%n;
                total_gas+=gas[j];
                total_cost+=cost[j];
                if(total_gas<total_cost)
                break;
                num++;
            }
            if(num==n)return i;
            else
            {
                i=i+num+1;
            }
        }
        return -1;

    }
};
